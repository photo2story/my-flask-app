rsiPeriod  = input.int(28, title="RSI 기간", minval=2, group="MRI & RSI")

i_fromMonth = input.int(defval=1, title='[백테스트] 시작 월', minval=1, maxval=12, group="매매 기간")

f_isDate() =>

// ========== [갭 & 트레일링] ==========
gap_filter      = input.float(0.6, "Filter Gaps", step=0.01, group="Gaps & Trailing")

// ========== [SMA Deadzone & PPO] ==========
sma_deadzone_pct    = input.float(0.25, title="SMA4h 중립 지대 (%)", minval=0.0, step=0.05, group="필터 조건") / 100
ppo_buy_threshold   = input.float(0.10, title="PPO BUY 임계값", group="필터 조건")
enable_flat_sar_buy = input.bool(true, title="Flat + SAR Bull 시 재진입 허용", group="필터 조건")
ppo_sell_threshold  = input.float(-0.20, title="PPO SELL 임계값", group="필터 조건")

// ========== [4시간 SMA] ==========
sma4h                = request.security(syminfo.tickerid, "240", ta.sma(close, 120))
sma4h_upper_deadzone = sma4h * (1 + sma_deadzone_pct)
sma4h_lower_deadzone = sma4h * (1 - sma_deadzone_pct)
is_in_sma4h_deadzone = close <= sma4h_upper_deadzone and close >= sma4h_lower_deadzone

// ========== [볼린저 밴드(일봉/환산/현재)] ==========
f_get_daily_bb(bb_len, bb_mult, _lookahead) =>
    daily_close_src = request.security(syminfo.tickerid, "D", close, lookahead=_lookahead)
    daily_basis = ta.sma(daily_close_src, bb_len)
    daily_dev  = bb_mult * ta.stdev(daily_close_src, bb_len)
    [daily_basis - daily_dev, daily_basis + daily_dev, daily_basis]
[daily_lower_bb_val, daily_upper_bb_val, daily_basis_val] = f_get_daily_bb(daily_bb_length, daily_bb_mult, barmerge.lookahead_on)

ma(source, length, _type) =>
    switch _type
        "SMA"        => ta.sma(source, length)
        "EMA"        => ta.ema(source, length)
        "SMMA (RMA)" => ta.rma(source, length)
        "WMA"        => ta.wma(source, length)
        "VWMA"       => ta.vwma(source, length)

var float[] daily_close_array = array.new_float(0)
var int previous_bar_time = 0
if barstate.isconfirmed and time != previous_bar_time
    array.push(daily_close_array, close)
    previous_bar_time := time
    if array.size(daily_close_array) > 48
        array.shift(daily_close_array)
var float daily_close = na
if array.size(daily_close_array) >= 20
    daily_close := array.avg(daily_close_array)
temp_basis = ma(na(daily_close) ? close : daily_close, daily_bb_length, maType)
temp_stdev = ta.stdev(na(daily_close) ? close : daily_close, daily_bb_length)
converted_basis = na(daily_close) ? daily_basis_val : temp_basis
converted_dev   = na(daily_close) ? 0 : daily_bb_mult * temp_stdev
converted_upper = converted_basis + converted_dev
converted_lower = converted_basis - converted_dev

bb_upper = use_converted_bb ? converted_upper : daily_upper_bb_val
bb_lower = use_converted_bb ? converted_lower : daily_lower_bb_val

current_bb_length = 20
current_bb_mult = 2.0
current_bb_basis = ta.sma(close, current_bb_length)
current_bb_dev = current_bb_mult * ta.stdev(close, current_bb_length)
current_bb_upper = current_bb_basis + current_bb_dev
current_bb_lower = current_bb_basis - current_bb_dev

bb_center = (current_bb_upper + current_bb_lower) / 2
bb_deadzone_upper = bb_center * (1 + bb_deadzone_pct / 100)
bb_deadzone_lower = bb_center * (1 - bb_deadzone_pct / 100)
is_in_bb_deadzone = close <= bb_deadzone_upper and close >= bb_deadzone_lower

// ========== [BUY Guard: Avoid entry near BB Top] ==========
avoid_buy_near_bb_upper = input.bool(true, title="Avoid BUY: Near BB Top", group="필터 조건")
buy_bb_upper_buffer_pct = input.float(0.5, title="BB Top Buffer(%)", minval=0.0, maxval=5.0, step=0.1, group="필터 조건")
near_bb_upper_30m = not na(current_bb_upper) and close >= current_bb_upper * (1 - buy_bb_upper_buffer_pct/100)
near_bb_upper_daily = not na(bb_upper) and close >= bb_upper * (1 - buy_bb_upper_buffer_pct/100)
avoid_buy_bb = avoid_buy_near_bb_upper and (near_bb_upper_30m or near_bb_upper_daily)

// ========== [HTF PPO & SAR (일봉 개념) 계산] ==========
// - 일봉(또는 선택한 HTF) 내에서 PPO 라인/시그널/히스토그램을 '같은 프레임'으로 계산
f_ppo_triple(_tf) =>
    request.security(syminfo.tickerid, _tf, () =>
        _ema12 = ta.ema(close, 12)
        _ema26 = ta.ema(close, 26)
        _ppo   = _ema26 == 0 ? 0 : ((_ema12 - _ema26) / _ema26) * 100
        _sig   = ta.ema(_ppo, 9)
        [_ppo, _sig, _ppo - _sig]
    , lookahead=barmerge.lookahead_off)
[ppo_htf_line, ppo_htf_signal, ppo_htf_hist] = f_ppo_triple(htf_ppo_tf)

// HTF SAR: 동일 프레임에서 SAR과 close를 비교/크로스 감지
htf_close = request.security(syminfo.tickerid, htf_sar_tf, close, lookahead=barmerge.lookahead_off)
sar_htf   = request.security(syminfo.tickerid, htf_sar_tf, ta.sar(sar_start, sar_increment, sar_maximum), lookahead=barmerge.lookahead_off)
sar_htf_is_bull = sar_htf < htf_close
sar_htf_cross_down = request.security(syminfo.tickerid, htf_sar_tf, ta.cross(ta.sar(sar_start, sar_increment, sar_maximum), close), lookahead=barmerge.lookahead_off)
sar_htf_cross_up   = request.security(syminfo.tickerid, htf_sar_tf, ta.cross(close, ta.sar(sar_start, sar_increment, sar_maximum)), lookahead=barmerge.lookahead_off)

// (참고) LTF(30m) PPO도 병행 계산(기존 로직 유지)
change = ta.change(close)
up = ta.rma(math.max(change, 0), rsiPeriod)
down = ta.rma(-math.min(change, 0), rsiPeriod)
rsi = down == 0 ? 100 : up == 0 ? 0 : 100 - (100 / (1 + up / down))

ema12      = ta.ema(close, 12)
ema26      = ta.ema(close, 26)
ppoLine    = ema26 == 0 ? 0 : (ema12 - ema26) / ema26 * 100
signalLine = ta.ema(ppoLine, 9)
ppoHist    = ppoLine - signalLine

// (교정) Daily PPO: LTF 값을 섞지 않고 '일봉 프레임' 내부 계산
[daily_ppoLine, daily_signalLine, daily_ppoHist] = f_ppo_triple("D")

// ========== [매매 비율/플래그] ==========
buyRatio  = mri < 0.28 ? 1.0 : mri < 0.33 ? 0.9 : mri < 0.38 ? 0.8 : mri < 0.43 ? 0.5 : 0.0
buyRatio := rsi < 30 ? math.min(buyRatio * 1.2, 1.0) : buyRatio
sellRatio = mri >= 0.43 and rsi > 45 ? 1.0 : mri >= 0.38 and rsi > 40 ? 0.8 : mri >= 0.33 and rsi > 35 ? 0.7 : mri >= 0.28 and rsi > 35 ? 0.6 : 0.0
sellRatio := ppoHist > 0 ? math.min(sellRatio, 0.3) : sellRatio

is_clear_above_sma4h = close > sma4h_upper_deadzone
is_clear_below_sma4h = close < sma4h_lower_deadzone

// ========== [타임프레임(ms)/시간 문자열] ==========
var int tf_ms = na
if barstate.isfirst
    current_tf = timeframe.period
    tf_ms := current_tf == "1" ? 60 * 1000 :
             current_tf == "5" ? 5 * 60 * 1000 :
             current_tf == "15" ? 15 * 60 * 1000 :
             current_tf == "30" ? 30 * 60 * 1000 :
             current_tf == "60" ? 60 * 60 * 1000 :
             current_tf == "240" ? 4 * 60 * 60 * 1000 :
             current_tf == "D" ? 24 * 60 * 60 * 1000 :
             30 * 60 * 1000

bar_kst_str   = str.format_time(time + 9*60*60*1000 + tf_ms, "yyyy-MM-dd HH:mm:ss")
alert_kst_str = str.format_time(timenow + 9*60*60*1000, "yyyy-MM-dd HH:mm:ss")

// ========== [갭 탐지] ==========
bearish_gap_size = low[2] - high
bullish_gap_size = low - high[2]
bearish_gap_stdev = ta.stdev(bearish_gap_size, 100)
bullish_gap_stdev = ta.stdev(bullish_gap_size, 100)

ppo_is_positive       = ppoHist > 0
touched_daily_lower_bb = low <= (request.security(syminfo.tickerid, "D", ta.sma(close, daily_bb_length)) - daily_bb_mult * request.security(syminfo.tickerid, "D", ta.stdev(close, daily_bb_length)))

// HTF 흐름 기준: 일봉(또는 선택 HTF)의 PPO/SAR 상태를 메인 필터로 사용
htf_trend_ok_for_long = use_htf_flow ? (ppo_htf_hist >= 0 and sar_htf_is_bull) : (daily_ppoHist >= 0)
htf_ppo_trend_down    = use_htf_flow ? (ppo_htf_hist < 0) : (daily_ppoHist < 0)

// 기존 보조 지표(4h SMA 등)
sma_trend_up = not na(sma4h) and not na(sma4h[10]) and not na(sma4h[5]) and not na(sma4h[15]) and not na(sma4h[20]) and not na(sma4h[30]) and not na(sma4h[1]) and not na(sma4h[2]) and
               sma4h > sma4h[10] and sma4h[5] > sma4h[15] and sma4h > sma4h[20] and sma4h > sma4h[30] and sma4h[1] > sma4h[2]

// ====== 진입/청산 로직에 HTF 흐름 반영 ======
indicators_ready = not na(ppoHist) and not na(daily_ppoHist) and not na(rsi) and not na(sma4h) and not na(bearish_gap_stdev) and not na(bullish_gap_stdev)

canBuyMidTerm   = indicators_ready and buyRatio > 0 and ppoHist > (ppo_buy_threshold + (conservative_mode ? 0.05 : 0)) and rsi < 70 and not is_in_sma4h_deadzone and sma_trend_up and strategy.position_size == 0 and inDateRange and htf_trend_ok_for_long and (not avoid_buy_bb)
canSellMidTerm  = indicators_ready and sellRatio > 0 and ppoHist < ppo_sell_threshold and rsi > 30 and is_clear_below_sma4h and not is_in_sma4h_deadzone and not sma_trend_up and inDateRange
bearish_gap_detected  = indicators_ready and high < low[2] and low[1] > high and bearish_gap_stdev > 0 and bearish_gap_size / bearish_gap_stdev > gap_filter
bullish_gap_detected  = indicators_ready and low > high[2] and high[1] < low and bullish_gap_stdev > 0 and bullish_gap_size / bullish_gap_stdev > gap_filter
canBuyShortTerm       = bullish_gap_detected and not is_in_sma4h_deadzone and strategy.position_size == 0 and inDateRange and htf_trend_ok_for_long and (not avoid_buy_bb)
canSellShortTerm      = bearish_gap_detected and is_clear_below_sma4h and not is_in_sma4h_deadzone and inDateRange
canBuyAggressiveReversal = indicators_ready and touched_daily_lower_bb and not is_in_sma4h_deadzone and strategy.position_size == 0 and inDateRange and htf_trend_ok_for_long and (not avoid_buy_bb)

// 보수적 모드일 때 추가 필터(HTF 흐름 유지)
if conservative_mode and not compat_r24_mode
    canBuyMidTerm := canBuyMidTerm and (ppoHist > ppoHist[1]) and is_clear_above_sma4h
    canBuyShortTerm := canBuyShortTerm and ppo_is_positive
    canBuyAggressiveReversal := canBuyAggressiveReversal and ppo_is_positive

// r24 호환 모드(HTF 흐름 필터만 유지, 나머지는 r24와 동일)
if compat_r24_mode
    canBuyMidTerm   := indicators_ready and buyRatio > 0 and ppoHist > ppo_buy_threshold and rsi < 70 and not is_in_sma4h_deadzone and sma_trend_up and strategy.position_size == 0 and inDateRange and htf_trend_ok_for_long and (not avoid_buy_bb)
    canSellMidTerm  := indicators_ready and sellRatio > 0 and ppoHist < ppo_sell_threshold and rsi > 30 and is_clear_below_sma4h and not is_in_sma4h_deadzone and not sma_trend_up and inDateRange
    bearish_gap_detected  := indicators_ready and high < low[2] and low[1] > high and bearish_gap_stdev > 0 and bearish_gap_size / bearish_gap_stdev > gap_filter
    bullish_gap_detected  := indicators_ready and low > high[2] and high[1] < low and bullish_gap_stdev > 0 and bullish_gap_size / bullish_gap_stdev > gap_filter
    canBuyShortTerm       := bullish_gap_detected and not is_in_sma4h_deadzone and strategy.position_size == 0 and inDateRange and htf_trend_ok_for_long and (not avoid_buy_bb)
    canSellShortTerm      := bearish_gap_detected and is_clear_below_sma4h and not is_in_sma4h_deadzone and inDateRange
    canBuyAggressiveReversal := indicators_ready and touched_daily_lower_bb and not is_in_sma4h_deadzone and strategy.position_size == 0 and inDateRange and htf_trend_ok_for_long and (not avoid_buy_bb)

// === 청산 조건(HTF 기준 반영) ===
// - SAR+PPO 청산: HTF SAR '하향 크로스' AND HTF PPO 히스토그램 음전
canExitBySARAndPPO = (use_htf_flow ? sar_htf_cross_down : ta.cross(request.security(syminfo.tickerid, "120", ta.sar(sar_start, sar_increment, sar_maximum)), close)) and htf_ppo_trend_down and strategy.position_size > 0
// - SMA120 교차(4h SMA 기준 기존 유지), BB Top & PPO는 기존 로직
sma120_cross_down = ta.cross(sma4h, close)
canExitBySMA120 = enable_sma_cross_exit and sma120_cross_down and not is_in_bb_deadzone and strategy.position_size > 0
canExitByBBTopAndPPO = enable_bb_top_exit and close >= bb_upper * 0.98 and (use_htf_flow ? (ppo_htf_hist < 0) : (daily_ppoHist < 0)) and not sma_trend_up and strategy.position_size > 0
